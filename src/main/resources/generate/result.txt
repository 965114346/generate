package {beanPackageName};

import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;
import {enumPackageName}.ResultEnum;

@Data
@AllArgsConstructor
@NoArgsConstructor
public class Result {
    
    /**
     * 成功/失败		默认：成功
     */
    private Boolean success = true;

    /**
     * 状态码	默认：200
     */
    private Integer code = 200;
    
    /**
     * 消息
     */
    private String msg;
    
    /**
     * 具体值
     */
    private Object data;
    
    /**
     * 有值返回成功
     */
    public static Result success(Object obj){
        return new Result(true, ResultEnum.SUCCESS.getCode(), ResultEnum.SUCCESS.getMessage(), obj);
    }

    /**
     * 无值成功返回
     */
    public static Result success(){
       return success(null);
    }

    /**
     * 错误返回
     */
    public static Result failed(String msg){
        return new Result(true, ResultEnum.ERROR.getCode(), msg, null);
    }

    /**
     * 错误返回
     */
    public static Result failed(){
        return failed(ResultEnum.ERROR.getMessage());
    }

    /**
     * 错误返回
     */
    public static Result error(){
        return new Result(true, ResultEnum.ERROR.getCode(), ResultEnum.ERROR.getMessage(), null);
    }
}
